我已经获取了文章的内容。PicX 是一个命令行图片压缩工具，支持以下主要功能：

1. **单图压缩** - 支持 PNG/JPG/TIFF 转 WebP，可调节质量、尺寸
2. **目录批量处理** - 递归处理整个文件夹，支持并行任务
3. **切片 (Tile)** - 为大图生成金字塔切片，附带 `manifest.json`
4. **预设模式** - blog/web/avatar/lossless 等预设
5. **环境检查** - `picx doctor` 诊断环境
6. **目标体积压缩** - `--target-size` 自动寻找最佳质量

GitHub: https://github.com/ingeniousfrog/picx
PyPI: `picx-image-optimizer`

### 核心功能模块代码示例

#### 1. 单图压缩 Tab

```python
import customtkinter as ctk
from picx import optimize_image  # 直接调用 PicX API
from tkinter import filedialog, messagebox

class SingleImageTab(ctk.CTkFrame):
    def __init__(self, master):
        super().__init__(master)
        
        # 文件选择
        self.file_path = ctk.StringVar()
        ctk.CTkEntry(self, textvariable=self.file_path, width=400).pack(pady=10)
        ctk.CTkButton(self, text="选择图片", command=self.select_file).pack()
        
        # 参数设置
        self.format_var = ctk.StringVar(value="webp")
        ctk.CTkOptionMenu(self, values=["webp", "jpg", "png"], 
                         variable=self.format_var).pack()
        
        self.quality = ctk.CTkSlider(self, from_=1, to=100, number_of_steps=100)
        self.quality.set(78)
        
        # 执行按钮
        ctk.CTkButton(self, text="开始压缩", command=self.process).pack(pady=20)
        
    def select_file(self):
        path = filedialog.askopenfilename(
            filetypes=[("Images", "*.png *.jpg *.jpeg *.tiff *.webp")]
        )
        self.file_path.set(path)
        
    def process(self):
        input_path = self.file_path.get()
        output_path = input_path.rsplit('.', 1)[0] + f".{self.format_var.get()}"
        
        try:
            result = optimize_image(
                input_path,
                output=output_path,
                format=self.format_var.get(),
                quality=int(self.quality.get()),
                # 其他参数...
            )
            messagebox.showinfo("成功", f"压缩完成！\n原大小: {result.original_size}\n新大小: {result.new_size}")
        except Exception as e:
            messagebox.showerror("错误", str(e))
```

#### 2. 目录批量处理 Tab

```python
import threading
from concurrent.futures import ThreadPoolExecutor

class BatchTab(ctk.CTkFrame):
    def __init__(self, master):
        super().__init__(master)
        self.tasks = []        # 任务列表
        self.executor = None
        self.is_running = False
        
    def start_batch(self):
        input_dir = self.input_dir.get()
        output_dir = self.output_dir.get()
        jobs = int(self.jobs.get())  # 并行数
        
        files = self.scan_images(input_dir)
        
        # 使用 ThreadPoolExecutor 并行处理
        with ThreadPoolExecutor(max_workers=jobs) as executor:
            futures = []
            for f in files:
                future = executor.submit(self.process_single, f, output_dir)
                futures.append(future)
            
            for future in futures:
                self.update_progress(future.result())
```

```python
class TileTab(ctk.CTkFrame):
    def __init__(self, master):
        super().__init__(master)
        
        # TIFF 专用处理
        ctk.CTkLabel(self, text="大图切片 (支持 TIFF)").pack()
        
        self.backend = ctk.StringVar(value="pyvips")
        ctk.CTkOptionMenu(self, values=["pyvips", "pillow"], 
                         variable=self.backend).pack()
        
        self.tile_size = ctk.CTkEntry(self, placeholder_text="1024")
        self.pyramid = ctk.CTkSwitch(self, text="生成金字塔层级")
        
    def generate_tiles(self):
        from picx import generate_tiles
        
        generate_tiles(
            self.input_path.get(),
            self.output_dir.get(),
            tile_size=int(self.tile_size.get()),
            format="webp",
            backend=self.backend.get(),
            pyramid=self.pyramid.get()
        )
        # 生成 manifest.json 预览
        self.preview_manifest()
```

#### 4. 环境诊断 (Doctor)

```python
class DoctorTab(ctk.CTkFrame):
    def __init__(self, master):
        super().__init__(master)
        
        ctk.CTkButton(self, text="🔍 运行环境检查", 
                     command=self.run_diagnosis).pack()
        
        self.result_text = ctk.CTkTextbox(self, width=500, height=300)
        self.result_text.pack()
        
    def run_diagnosis(self):
        # 调用 picx doctor 功能
        import subprocess
        result = subprocess.run(["picx", "doctor"], 
                              capture_output=True, text=True)
        self.result_text.insert("end", result.stdout)
```
